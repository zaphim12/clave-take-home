Three tables is a good approach for storing the data:
ORDERS, ORDER_ITEMS, ORDER_ITEM_OPTIONS

CREATE TABLE ORDERS (
  order_id VARCHAR(255) PRIMARY KEY,
  store_id VARCHAR(255) NOT NULL,
  fulfillment_method VARCHAR(100),
  created_at TIMESTAMP,
  tip DECIMAL(10, 2),
  tax DECIMAL(10, 2),
  total DECIMAL(10, 2),
  provider VARCHAR(100)
);

CREATE TABLE ORDER_ITEMS (
  order_item_id uuid primary key default gen_random_uuid(),
  order_id VARCHAR(255) NOT NULL,
  item_id VARCHAR(255) NOT NULL,
  name VARCHAR(255),
  quantity INT,
  unit_price DECIMAL(10, 2),
  special_instructions VARCHAR(255),
  category VARCHAR(100),
  canonical_item_id uuid,
  canonical_category_id uuid,
  FOREIGN KEY (order_id) REFERENCES ORDERS(order_id),
  FOREIGN KEY (canonical_item_id) REFERENCES CANONICAL_ITEMS(id),
  FOREIGN KEY (canonical_category_id) REFERENCES CANONICAL_CATEGORIES(id)
);

CREATE TABLE ORDER_ITEM_OPTIONS (
  order_item_options_id uuid primary key default gen_random_uuid(),
  order_id VARCHAR(255) NOT NULL,
  order_item_id INT NOT NULL,
  item_id VARCHAR(255) NOT NULL,
  name VARCHAR(255),
  price DECIMAL(10, 2),
  FOREIGN KEY (order_id) REFERENCES ORDERS(order_id),
  FOREIGN KEY (order_item_id) REFERENCES ORDER_ITEMS(order_item_id)
);

Plus we can use some helper tables for matching inconsistently named items and categories:
CREATE TABLE CANONICAL_ITEMS (
  id uuid primary key default gen_random_uuid(),
  canonical_name text not null,
  normalized_name text not null unique
);

CREATE TABLE ITEM_MAPPINGS (
  raw_name text not null,
  normalized_raw_name text not null,
  canonical_item_id uuid references canonical_items(id),
  mapping_method text,
  confidence numeric,
  unique(normalized_raw_name)
);

CREATE TABLE CANONICAL_CATEGORIES (
  id uuid primary key default gen_random_uuid(),
  canonical_name text not null,
  normalized_name text not null unique
);

CREATE TABLE CATEGORY_MAPPINGS (
  raw_category text not null,
  normalized_raw_category text not null unique,
  canonical_category_id uuid references canonical_categories(id),
  mapping_method text,
  confidence numeric
);

These tables are populated from the included json files using the following data. 
We aren't storing all possible data. Instead we'll just store the data which is present across all the json files.
This will still give us data on each order's location, items, categories, price, tax, tip, and fulfillment method

doordash_orders.json:
ORDERS:
- order_id = external_delivery_id
- store_id = store_id
- fulfillment_method = order_fulfillment_method
- created_at = created_at
- tip = dasher_tip
- tax = tax_amount
- total = total_charged_to_consumer
- provider = (explicit-string) "Doordash"

ORDER_ITEMS:
- order_id = external_delivery_id
- order_item_id = generated-by-SQL
- item_id = item_id
- name = name
- quantity = quantity
- unit_price = unit_price
- special_instructions = special_instructions
- category = category

ORDER_ITEM_OPTIONS:
- order_item_options_id = <generated by SQL>
- order_id = external_delivery_id
- order_item_id = <generated by SQL>
- item_id = item_id
- name = name
- price = price

toast_pos_export.json:
ORDERS:
- order_id = guid
- store_id = restaurantGuid
- fulfillment_method = diningOption.behavior
- created_at = checks.openedDate
- tip = payments.tipAmount
- tax = payments.taxAmount
- total = payments.totalAmount
- provider = (explicit-string) "Toast"

ORDER_ITEMS:
- order_id = guid
- order_item_id = <generated by SQL>
- item_id = selections.guid
- name = displayName
- quantity = quantity
- unit_price = price
- category = itemGroup.name

ORDER_ITEMS_OPTIONS:
- order_item_options_id = <generated by SQL>
- order_id = guid
- order_item_id = <generated by SQL>
- item_id = selections.guid
- name = displayName
- price = price

square:
ORDER:
